import { Meta, IconGallery, IconItem } from "@storybook/blocks";
import { bluePalette } from "../theme/light/light-color-palette";
import * as Icons from "../custom-icons";
import { Tooltip } from "..";
import { useState } from "react";
import { TextField, IconButton } from "@mui/material";

<Meta title="Foundations/Iconography" />

export default function Iconography() {
  const [searchTerm, setSearchTerm] = useState("");

  const filteredIcons = Object.entries(Icons).filter(([name]) =>
    name.toLowerCase().includes(searchTerm.toLowerCase())
  );

  return (
    <>
      <TextField
        label="Search icons"
        value={searchTerm}
        onChange={(e) => setSearchTerm(e.target.value)}
        fullWidth
        sx={{
          marginBottom: "40px",
        }}
      />
      <IconGallery>
        {filteredIcons.map(([name, Icon], index) => (
          <IconItem key={index} name={name}>
            <Tooltip title={name}>
              <IconButton>
                <Icon fill={bluePalette[500]} />
              </IconButton>
            </Tooltip>
          </IconItem>
        ))}
      </IconGallery>
    </>
  ); 
}
